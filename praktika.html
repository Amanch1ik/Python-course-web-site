Проект 1 — Калькулятор

Создадим простой консольный калькулятор, который выполняет основные арифметические операции. Этот проект объединяет функции, условия и обработку ввода.

calculator.py 
def сложить(a, b):
    return a + b

def вычесть(a, b):
    return a - b

def умножить(a, b):
    return a * b

def разделить(a, b):
    if b == 0:
        return "Ошибка: деление на ноль!"
    return a / b

def калькулятор(a, операция, b):
    операции = {
        "+": сложить,
        "-": вычесть,
        "*": умножить,
        "/": разделить
    }
    if операция not in операции:
        return "Неизвестная операция"
    функция = операции[операция]
    return функция(a, b)

# Тестирование
примеры = [
    (10, "+", 5),
    (10, "-", 3),
    (4,  "*", 7),
    (15, "/", 4),
    (9,  "/", 0),
]

for a, оп, b in примеры:
    рез = калькулятор(a, оп, b)
    print(f"{a} {оп} {b} = {рез}")
Что использовано: функции, словарь как диспетчер операций, цикл for, обработка ошибки деления на ноль, f-строки для форматирования вывода.

Проект 2 — Угадай число

Программа загадывает случайное число от 1 до 100. Игрок вводит попытки, и программа подсказывает «больше» или «меньше». Используем модуль random из стандартной библиотеки.

guess_game.py 
import random

def игра_угадай_число():
    загаданное = random.randint(1, 100)
    попыток = 0
    максимум_попыток = 7

    print("Я загадал число от 1 до 100.")
    print(f"У вас {максимум_попыток} попыток.\n")

    while попыток < максимум_попыток:
        попыток += 1
        осталось = максимум_попыток - попыток + 1

        try:
            попытка = int(input(f"Попытка {попыток}/{максимум_попыток}: "))
        except ValueError:
            print("Введите целое число!\n")
            попыток -= 1
            continue

        if попытка == загаданное:
            print(f"\nПравильно! Вы угадали за {попыток} попыток!")
            return
        elif попытка < загаданное:
            print(f"Больше. Осталось попыток: {осталось - 1}\n")
        else:
            print(f"Меньше. Осталось попыток: {осталось - 1}\n")

    print(f"\nПопытки закончились. Загаданное число: {загаданное}")

Проект 3 — Менеджер задач

Консольное приложение для управления списком дел. Демонстрирует работу со словарями, списками и функциями в связке.

todo.py 
задачи = []

def добавить_задачу(название):
    задача = {"название": название, "выполнено": False}
    задачи.append(задача)
    print(f'Добавлено: "{название}"')

def выполнить_задачу(номер):
    if 0 <= номер < len(задачи):
        задачи[номер]["выполнено"] = True
        print(f'Выполнено: "{задачи[номер]["название"]}"')
    else:
        print("Задача не найдена")

def показать_задачи():
    if not задачи:
        print("Список задач пуст")
        return
    print("\n--- Список задач ---")
    for i, з in enumerate(задачи):
        статус = "[+]" if з["выполнено"] else "[ ]"
        print(f"  {i}. {статус} {з['название']}")
    выполнено = sum(1 for з in задачи if з["выполнено"])
    print(f"Выполнено: {выполнено}/{len(задачи)}\n")

# Демонстрация

добавить_задачу("Сделать проект-калькулятор")
