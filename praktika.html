Квадратное уравнение

Решите квадратное уравнение ax² + bx + c = 0 через дискриминант. Обработайте все три случая.

Посмотреть решение
task12.py
import math

def квадратное(a, b, c):
    д = b**2 - 4*a*c
    if д > 0:
        x1 = (-b + math.sqrt(д)) / (2*a)
        x2 = (-b - math.sqrt(д)) / (2*a)
        print(f"Два корня: x1={x1:.2f}, x2={x2:.2f}")
    elif д == 0:
        x = -b / (2*a)
        print(f"Один корень: x={x:.2f}")
    else:
        print("Нет действительных корней")

квадратное(1, -5, 6)   # x²-5x+6=0
квадратное(1, -2, 1)   # x²-2x+1=0
квадратное(1,  1, 5)   # x²+x+5=0

    n = abs(n)

        n //= 10
    return итог

def простые_до(n):
    числа = []
    for число in range(2, n + 1):

Сортировка пузырьком

Реализуйте алгоритм сортировки пузырьком. Не используйте встроенные функции sort() и sorted().

Посмотреть решение
task30.py
def пузырёк(список):
    список = список.copy()
    n = len(список)
    for i in range(n):
        for j in range(0, n - i - 1):
            if список[j] > список[j + 1]:
                список[j], список[j + 1] = список[j + 1], список[j]
    return список

числа = [64, 34, 25, 12, 22, 11, 90]
print(f"Исходный:       {числа}")
print(f"Отсортированный: {пузырёк(числа)}")

def частота_букв(текст):
    счётчик = {}
    for буква in текст.lower():
        if буква.isalpha():
            счётчик[буква] = счётчик.get(буква, 0) + 1
    return dict(sorted(счётчик.items(), key=lambda x: x[1], reverse=True))

def в_двоичную(n):
    if n == 0:
        return "0"
    биты = []
    while n > 0:
        биты.append(str(n % 2))
        n //= 2
    return "".join(reversed(биты))

for n in [0, 1, 5, 10, 255]:

Быстрое возведение в степень

Реализуйте рекурсивное возведение числа в степень за O(log n) шагов с использованием метода быстрого возведения.

Посмотреть решение
task44.py
def степень(основание, показатель):
    if показатель == 0:
        return 1
    if показатель < 0:
        return 1 / степень(основание, -показатель)
    if показатель % 2 == 0:          # чётная степень
        половина = степень(основание, показатель // 2)
        return половина * половина   # один вызов вместо двух
    return основание * степень(основание, показатель - 1)

for осн, пок in [(2, 10), (3, 5), (5, 0), (2, -3)]:
    print(f"{осн}^{пок} = {степень(осн, пок)}")

def вычесть(a, b): return a - b

def разделить(a, b):
    if b == 0:

    return a / b

def калькулятор(a, операция, b):

Проект 2 — Угадай число

Программа загадывает случайное число от 1 до 100. Игрок вводит попытки, а программа подсказывает «больше» или «меньше». Используется модуль random.

guess_game.py 
import random

def игра_угадай_число():
    загаданное = random.randint(1, 100)
    попыток = 0
    максимум = 7

    print("Я загадал число от 1 до 100.")
    print(f"У вас {максимум} попыток.\n")

    while попыток < максимум:
        попыток += 1
        try:
            попытка = int(input(f"Попытка {попыток}/{максимум}: "))
        except ValueError:
            print("Введите целое число!\n")
            попыток -= 1
            continue

        if попытка == загаданное:
            print(f"\nПравильно! Угадали за {попыток} попыток!")
            return
        elif попытка < загаданное:
            print("Больше.\n")
        else:
            print("Меньше.\n")

    print(f"\nПопытки закончились. Загаданное число: {загаданное}")
